\begin{problem}{6.5.4}
  Let $F:[0,1) \to [0,1)$ be the tripling map. Show that $F$ is transitive
  and that its periodic points are dense in $[0, 1)$.
\end{problem}

\begin{proof}
  Note that
  \begin{align*}
    F(x):=
    \begin{cases}
      3x    & \text{if $x \in [0, 1/3)$ }\\
      3x -1 & \text{if $x \in [1/3, 2/3)$ }\\
      3x -2 & \text{if $x \in [2/3, 1)$ }
    \end{cases}
  \end{align*}
\end{proof}
